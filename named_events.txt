buyLeather
	text You enter a trading village.
	option Sell wood
		test wood >= 20
		effects wood - 20 $ + 5
	option Buy leather
		test $ >= 5
		effects leather + 15 $ - 5
	option Recruit War Leader
		test $ >= 8 warLeader = 0
		effects $ - 8 warLeader = 1 strength + 3
	option Recruit Engineer
		test $ >= 8 engineer = 0
		effects $ - 8 engineer = 1
	option Trade Valuables
		effects $ + 2
	option Leave

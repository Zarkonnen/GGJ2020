upkeep
	effects mvCooldown - 1 food + food_upkeep water + water_upkeep hunger + hunger_upkeep thirst + thirst_upkeep

buyLeather
	text You enter a trading village.
	option Sell wood
		test wood >= 20
		effects wood - 20 $ + 5
	option Buy leather
		test $ >= 5
		effects leather + 15 $ - 5
	option Recruit War Leader
		test $ >= 8 warLeader = 0
		effects $ - 8 warLeader = 1
	option Don't
